import pandas as pd

# Function to generate dates for a given year, excluding Feb 28 and Feb 29
def generate_dates(year):
    dates = pd.date_range(start=f"{year}-01-01", end=f"{year}-12-31", freq='D')
    return dates[~((dates.month == 2) & (dates.day.isin([28, 29])))]

# Generate date series for all years
dates_2022 = pd.Series(generate_dates(2022))
dates_2023 = pd.Series(generate_dates(2023))
dates_2024 = pd.Series(generate_dates(2024))
dates_2025 = pd.Series(generate_dates(2025))

# Determine the maximum length
max_len = builtins.max(len(dates_2022), len(dates_2023), len(dates_2024), len(dates_2025))

# Reindex to make all series the same length
dates_2022 = dates_2022.reindex(range(max_len))
dates_2023 = dates_2023.reindex(range(max_len))
dates_2024 = dates_2024.reindex(range(max_len))
dates_2025 = dates_2025.reindex(range(max_len))

spark_model = f"""
with weekly_sales_info as (
select 
    week_number,
    txn.model_name,
    region_name,
    txn.material_group_name,
    txn.category_name,
    round(sum(amount),2) as sales,
    count(distinct transaction_id) as num_txns,
    sum(txn.quantity) as quantity,
    round(sum(amount)/sum(txn.quantity),2) as avg_unit_price,
    count(distinct store_id) as num_stores,
    count(distinct case when day_month in (31,30,1,2,3,4,5) then business_day end) as pay_days,
    vol.new_units as units,
    round(sum(ean_volume * txn.quantity),3) as total_volume,
    round(sum(amount)/sum(ean_volume * txn.quantity),5) as per_vol_price,
    concat_ws(',', collect_set(concat(vol.material_id,'-',vol.material_name))) as material_set
from {txn_data_raw_dg_poc_name} txn
left join {Base_table} vol ON txn.material_id = vol.material_id and txn.ean = vol.ean 
Where 
  vol.valid_price_change_points >= 5 
  AND vol.history_weeks_with_sales >= 52
  AND txn.material_group_name = '{selected_mg}'
  AND vol.material_group_name = '{selected_mg}'
GROUP BY 
    week_number,
    txn.model_name,
    region_name,
    txn.material_group_name,
    txn.category_name,
    vol.new_units
),

holiday_info as (
  select holiday_week, count(holiday_date) as num_holidays

        cast(holiday_date as date) as holiday_date,

import openmeteo_requests
import pandas as pd
import numpy as np
import requests_cache
from retry_requests import retry

# Setup the Open-Meteo API client
cache_session = requests_cache.CachedSession('.cache', expire_after=-1)
retry_session = retry(cache_session, retries=5, backoff_factor=0.2)
openmeteo = openmeteo_requests.Client(session=retry_session)

# UAE cities and their coordinates
cities = {
    "Abu Dhabi": (24.4539, 54.3773),
    "Dubai": (25.276987, 55.296249),
    "Sharjah": (25.3463, 55.4209),
    "Al Ain": (24.1302, 55.8023)
}

# Common API parameters
base_params = {
    "start_date": "2022-10-02",
    "end_date": "2025-10-29",
    "hourly": ["temperature_2m", "rain"]
}

# Store data for all cities
all_data = []

for city, (lat, lon) in cities.items():
    print(f"Fetching data for {city}...")
    params = base_params.copy()
    params.update({"latitude": lat, "longitude": lon})

    # API Call
    responses = openmeteo.weather_api("https://archive-api.open-meteo.com/v1/archive", params=params)
    response = responses[0]
    
    hourly = response.Hourly()
    temperature = hourly.Variables(0).ValuesAsNumpy()
    rain = hourly.Variables(1).ValuesAsNumpy()
    
    timestamps = pd.date_range(
        start=pd.to_datetime(hourly.Time(), unit="s", utc=True),
        end=pd.to_datetime(hourly.TimeEnd(), unit="s", utc=True),
        freq=pd.Timedelta(seconds=hourly.Interval()),
        inclusive="left"
    )

    df = pd.DataFrame({
        "datetime": timestamps,
        "temperature_2m": temperature,
        "rain": rain
    })
    df["city"] = city
    df["date"] = df["datetime"].dt.date
    df["week"] = df["datetime"].dt.to_period("W").apply(lambda r: r.start_time.strftime('%Y-W%W'))
    df["month"] = df["datetime"].dt.to_period("M").apply(lambda r: r.start_time.strftime('%Y-M%m'))

    all_data.append(df)

# Combine data from all cities

def get_dg_seasonality_idx(df_final_mds_base):

    import pandas as pd
    import numpy as np

    df_final_mds_base['year'] = df_final_mds_base['week_number'].str[:4]
    df_final_mds_base['week_number_val'] = df_final_mds_base['week_number'].str[-3:]

    # Include region in pivot index
    df_seasonality_idx = df_final_mds_base.pivot_table(
        index=['material_group_name', 'region_name', 'week_number_val'],
        columns='year',
        values='quantity'
    )

    df_seasonality_idx.columns = [f'year_{str(col)[-2:]}' for col in df_seasonality_idx.columns]
    df_seasonality_idx.reset_index(inplace=True)

    year_columns = [col for col in df_seasonality_idx.columns if col.startswith('year_')]

    def calculate_overall_avg(group):
        all_values = group[year_columns].values.flatten()
        all_values = all_values[~np.isnan(all_values)]
        return np.mean(all_values) if len(all_values) > 0 else np.nan

    # Group by material and region_name
    overall_avg = df_seasonality_idx.groupby(['material_group_name', 'region_name']).apply(calculate_overall_avg)

    # Map overall average to each row
    df_seasonality_idx['overall_avg_qty'] = df_seasonality_idx.set_index(['material_group_name', 'region_name']).index.map(overall_avg)

    df_seasonality_idx['yearly_avg_qty'] = df_seasonality_idx[year_columns].mean(axis=1, skipna=True)

    df_seasonality_idx['seasonality_idx'] = np.round(
        df_seasonality_idx['yearly_avg_qty'] / df_seasonality_idx['overall_avg_qty'], 3
    )

    # Final output: include region_name
    final_cols = ['material_group_name', 'week_number_val', 'region_name', 'seasonality_idx']
    df_seasonality_idx_final = df_seasonality_idx[final_cols].copy()

    return df_seasonality_idx_final

pay_day_cal_code = f"""

-- Step 1: Assign pay-day bucket and index per day
WITH base_with_calendar AS (
    select 
        region_name,
        business_day,
        week_number,
        day_month as day_of_month,
        model_name,
        round(sum(amount)) as sales,
        sum(quantity) as quantity,
        round(sum(amount)/sum(quantity),2) as unit_price_derived,
        CASE
            WHEN day_of_month BETWEEN 1 AND 5  THEN 'Pay-day Spike'
            WHEN day_of_month BETWEEN 6 AND 10 THEN 'Month Above Avg'
            WHEN day_of_month BETWEEN 11 AND 25 THEN 'Mid-month Below Avg'
            WHEN day_of_month BETWEEN 26 AND 31 THEN 'Month Above Avg'
        END AS pay_day_bucket
    from {txn_data_raw_dg_poc_name}
    where material_group_name = '{selected_mg}'
    group by 
        region_name,
        business_day,
        week_number,
        day_month,
        model_name
),

-- Step 2: Overall avg sales per model-region (daily level)
overall_avg AS (
    SELECT 
        model_name,
        region_name,
        AVG(sales) AS avg_daily_sales
    FROM base_with_calendar
    GROUP BY model_name, region_name
),

-- Step 3: Dynamic bucket indices per product-region
bucket_index_calc AS (
    SELECT
        d.model_name,
        d.region_name,
        d.pay_day_bucket,
        round(AVG(d.sales) / AVG(o.avg_daily_sales),2) AS bucket_index
    FROM base_with_calendar d
    JOIN overall_avg o ON d.model_name = o.model_name AND d.region_name = o.region_name
    GROUP BY 
        d.model_name, 
        d.region_name, 
        d.pay_day_bucket
),

-- Step 4: Attach dynamic bucket index back to each daily record
model_region_week_index AS (
  SELECT
    d.model_name,
    d.region_name,
    d.week_number,
    ROUND(avg(b.bucket_index),3) as pay_day_index
  FROM base_with_calendar d
  JOIN bucket_index_calc b
    ON d.pay_day_bucket = b.pay_day_bucket
      AND d.model_name = b.model_name
      AND d.region_name = b.region_name
    GROUP BY d.model_name,
        d.region_name,
        d.week_number
)

select 
  *
from model_region_week_index

"""

pay_day_cal = spark.sql(pay_day_cal_code)

    AS quantity_lag_2,
    ROUND(
      (LAG(quantity, 1) OVER (PARTITION BY material_id ORDER BY week) +
       LAG(quantity, 2) OVER (PARTITION BY material_id ORDER BY week)) / 2.0, 2
    ) AS quantity_ma_2w_excl_current,

    ROUND(AVG(seasonality_idx) OVER (
      PARTITION BY material_id
      ORDER BY week
      ROWS BETWEEN 2 PRECEDING AND 1 PRECEDING
    ), 2) AS seasonality_2w_ma,

    ROUND(AVG(seasonality_idx) OVER (
      PARTITION BY material_id
      ORDER BY week
      ROWS BETWEEN 3 PRECEDING AND 1 PRECEDING
    ), 2) AS seasonality_3w_ma,

    ROUND(AVG(seasonality_idx) OVER (
      PARTITION BY material_id
      ORDER BY week
      ROWS BETWEEN 5 PRECEDING AND 1 PRECEDING
    ), 2) AS seasonality_5w_ma,

    -- Price moving averages --

      PARTITION BY material_id ORDER BY week

      PARTITION BY material_id ORDER BY week 
      ROWS BETWEEN 3 PRECEDING AND CURRENT ROW
    ) AS avg_unit_price_derived_ma_4W,

    -- Quantity vs price ratios --
    unit_price_derived 
      / NULLIF(AVG(unit_price_derived) OVER (
           PARTITION BY material_id ORDER BY week ROWS BETWEEN 3 PRECEDING AND CURRENT ROW
         ),0)
      AS price_ratio_ma_4w_derived,

    unit_price_derived 
      / NULLIF(AVG(unit_price_derived) OVER (
           PARTITION BY material_id ORDER BY week ROWS BETWEEN 1 PRECEDING AND CURRENT ROW
         ),0)

    -- Competitor price ratio re-used as-is from cte_raw

    SUM(brand_competitors_total_discount) AS brand_competitors_total_discount,
