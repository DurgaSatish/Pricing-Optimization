def get_model(name, params):
    """
    Returns a regression model based on the given name and parameters.
    
    Args:
        name (str): Name of the model (e.g., 'RandomForestRegressor', 'LogLinear', etc.).
        params (dict): Parameters to pass to the model.

    Returns:
        model (object): Initialized regression model or 'None' if the name is invalid.
    """

    if name == 'RandomForestRegressor':
        model = RandomForestRegressor(random_state=42, **params)
    elif name == 'AdaBoostRegressor':
        model = AdaBoostRegressor(random_state=42, **params)
    elif name == 'CatBoostRegressor':
        model = CatBoostRegressor(verbose=0, random_state=42, **params)
    elif name == 'XGBRegressor':
        model = XGBRegressor(random_state=42, **params)
    elif name in ('LogLinear', 'LogLogLinear', 'LinearLog', 'LinearRegression'):
        model = LinearRegression(**params)
    elif name == 'ElasticNet':
        model = ElasticNet(random_state=42, **params)
    else:
        model = None 

    return model
def model_data_prep(data,model_filter=None, 
                   group_columns=['region_name', 'material_group_name','feature_set_name', 'model_name']):
    """
    Prepares data for modeling by:
    1. Filtering for specific model if provided
    2. Preserving grouping columns (region, week, model_name)
    3. Dropping other unnecessary identifier columns
    4. Normalizing specified numerical columns within each group
    5. Removing columns with constant values within groups

    Args:
        data (pd.DataFrame): The input dataset
        x_other_var (list): List of numerical columns to normalize
        model_filter (str or list): Specific category(ies) to filter for. If None, uses all model
        group_columns (list): Columns to group by for fitting models

    Returns:
        pd.DataFrame: The transformed dataset ready for modeling
    """
    
    # Filter by category if specified
    if model_filter is not None:
        if isinstance(model_filter, str):
            data = data[data['model_name'] == model_filter].copy()
            print(f"Filtered data for category: {model_filter}")
        elif isinstance(model_filter, list):
            data = data[data['model_name'].isin(model_filter)].copy()
            print(f"Filtered data for model: {model_filter}")
        
        if len(data) == 0:
            raise ValueError(f"No data found for model filter: {model_filter}")
        
        print(f"Data shape after model filter: {data.shape}")
    
    # Separate group columns from the data
    group_data = data[group_columns].copy()
    
    # Define columns to drop (excluding the group columns we want to keep)
    identifier_columns_to_drop = ['material_group_name','category_name',
                                  'price_column_used', 'quantity_column_used', 'seasonality_used']
    
    # Remove group columns from the drop list to avoid dropping them
    identifier_columns_to_drop = [col for col in identifier_columns_to_drop 
                                  if col not in group_columns]
    
    # Get feature columns (everything except group columns and columns to drop)
    feature_columns = [col for col in data.columns 
                      if col not in group_columns and col not in identifier_columns_to_drop]
    
    # Create feature data
    feature_data = data[feature_columns].copy()
    
    # Normalize numerical columns within each group
    normalize_feature = x_other_var + x_competitors  +['seasonality']
    columns_to_normalize = [col for col in normalize_feature if col in feature_columns]
    print(columns_to_normalize)
    print("Available feature columns:", feature_columns)
    data_normalized = pd.concat([group_data, feature_data], axis=1)
    

    final_data = []
    
    for feature_set, fs_data in data_normalized.groupby('feature_set_name'):
        # Find non-constant columns for this feature set
        non_constant_mask = (fs_data != fs_data.iloc[0]).any()
        
        # Always keep group columns even if they're constant within a group
        for col in group_columns:
            if col in non_constant_mask.index:
                non_constant_mask[col] = True
        
        # Get non-constant columns
        columns_to_keep = non_constant_mask[non_constant_mask].index.tolist()
        
        # Report removed columns
        constant_columns = set(fs_data.columns) - set(columns_to_keep)
        if constant_columns:
            print(f"Feature set '{feature_set}' - Removed constant columns: {list(constant_columns)}")
        
        # Keep only non-constant columns
        fs_data_cleaned = fs_data[columns_to_keep]
        final_data.append(fs_data_cleaned)
    
    # Combine all feature sets
    result = pd.concat(final_data, ignore_index=True)
    
    # Ensure all group columns are present in the final result
    for col in group_columns:
        if col not in result.columns:
            result[col] = data[col]
    result = result.reset_index(drop=True)
    return result

def get_params_range(trial, name):
    if name == 'RandomForestRegressor':
        params = {
            'n_estimators': trial.suggest_int('n_estimators', 50, 500),
            'max_depth': trial.suggest_int('max_depth', 3, 20),
            'min_samples_split': trial.suggest_int('min_samples_split', 2, 20),
            'min_samples_leaf': trial.suggest_int('min_samples_leaf', 1, 20),
            'max_features': trial.suggest_categorical('max_features', [1.0, 'sqrt', 'log2']),
            'bootstrap': trial.suggest_categorical('bootstrap', [True, False]),
        }
        if params['bootstrap']:
            params['oob_score'] = trial.suggest_categorical('oob_score', [True, False])
        else:
            params['oob_score'] = False  # Force False if no bootstrapping

    elif name == 'AdaBoostRegressor':
        params = {
            'n_estimators': trial.suggest_int('n_estimators', 50, 500),
            'learning_rate': trial.suggest_float('learning_rate', 0.01, 1.0, log=True),
            'loss': trial.suggest_categorical('loss', ['linear', 'square', 'exponential'])
        }

    elif name == 'CatBoostRegressor':
        params = {
            'iterations': trial.suggest_int('iterations', 100, 1000),
            'learning_rate': trial.suggest_float('learning_rate', 0.01, 1.0, log=True),
            'depth': trial.suggest_int('depth', 4, 16),
            'l2_leaf_reg': trial.suggest_float('l2_leaf_reg', 1e-2, 10.0, log=True),
            'bagging_temperature': trial.suggest_float('bagging_temperature', 0.0, 1.0),
            'border_count': trial.suggest_int('border_count', 32, 255),
            'random_strength': trial.suggest_float('random_strength', 0.0, 10.0),
            'grow_policy': trial.suggest_categorical('grow_policy', ['SymmetricTree', 'Depthwise', 'Lossguide']),
        }

    elif name == 'XGBRegressor':
        params = {
            'n_estimators': trial.suggest_int('n_estimators', 50, 500),
            'max_depth': trial.suggest_int('max_depth', 3, 10),
            'learning_rate': trial.suggest_float('learning_rate', 1e-4, 0.1, log=True),
            'subsample': trial.suggest_float('subsample', 0.5, 1.0),
            'colsample_bytree': trial.suggest_float('colsample_bytree', 0.5, 1.0),
            'gamma': trial.suggest_float('gamma', 0, 5),
            'reg_alpha': trial.suggest_float('reg_alpha', 1e-8, 1.0, log=True),
            'reg_lambda': trial.suggest_float('reg_lambda', 1e-8, 1.0, log=True),
        }
    elif name == 'ElasticNet':
        params = {
            'alpha': trial.suggest_float('alpha', 1e-4, 10.0, log=True),
            'l1_ratio': trial.suggest_float('l1_ratio', 0.0, 1.0),
            'fit_intercept': trial.suggest_categorical('fit_intercept', [True, False]),
            'max_iter': trial.suggest_int('max_iter', 500, 5000),
        }
    elif name in ['LinearRegression', 'LogLinear', 'LinearLog', 'LogLogLinear']:
        params = {
            'fit_intercept': trial.suggest_categorical('fit_intercept', [True, False]),
        }
    else:
        params = 'None'
    
    return params
def optuning(df_mds, df_region_sku_sampled,test_size,name,n_trials):
    """
    Optimize hyperparameters for a given model using Optuna.
    
    Args:
        df_mds: Full dataset
        df_region_sku_sampled: Sampled region-material combinations
        name: Model name (e.g., 'RandomForestRegressor')
    
    Returns:
        DataFrame with averaged best hyperparameters
    """
    
    best_params_df = pd.DataFrame()
    successful_combos = 0
    failed_combos = 0
    
    for i in range(len(df_region_sku_sampled)):
        try:
            df_mds_optuna = df_region_sku_sampled.iloc[i:i+1].copy()
            df_mds_optuna = df_mds_optuna.merge(
                df_mds, 
                on=['region_name', 'model_name'], 
                how='left'
            )
            if len(df_mds_optuna) < 10:
                failed_combos += 1
                continue
                
            columns_to_drop = ['quantity', 'week', 'week_number', 'region_name', 'model_name', 
                              'feature_set_name', 'category_name','total_volume',
                              'material_group_name', 'price_column_used', 'quantity_column_used', 
                              ]
            
            X = df_mds_optuna.drop(columns=[col for col in columns_to_drop if col in df_mds_optuna.columns])
            X = X.select_dtypes(include=[np.number])
            
            # Skip if no numeric features
            if X.shape[1] == 0:
                print(f"  No numeric features found for combination {i+1}")
                failed_combos += 1
                continue
            
            y = df_mds_optuna['total_volume']
            
            # Train-test split
            X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=test_size, random_state=42,shuffle = True)
            print(X_train.columns)
            normalize_features = x_other_var + x_competitors + ['seasonality']
            scaler = MinMaxScaler()
            X_train_scaled = X_train.copy()
            X_train_scaled[normalize_features] = scaler.fit_transform(X_train[normalize_features])

            # Apply the same scaler to test data
            X_test_scaled = X_test.copy()
            X_test_scaled[normalize_features] = scaler.transform(X_test[normalize_features])
            
            def objective(trial):
                try:
                    params = get_params_range(trial, name)
                    # if name == 'Linear':
                    #     X_train_scaled = X_train.copy()
                    #     y_train = y_train.copy()

                    # if name == 'LogLinear':        # ln(Y) = β0 + β1 X
                    #     X_train_scaled = X_train.copy()
                    #     y_train = np.log1p(y_train)

                    # elif name == 'LinearLog':        # Y = β0 + β1 ln(X)
                    #     X_train_scaled = np.log1p(X_train)
                    #     y_train = y_train.copy()

                    # elif name == 'LogLogLinear':     # ln(Y) = β0 + β1 ln(X)
                    #     X_train_scaled = np.log1p(X_train)
                    #     y_train = np.log1p(y_train)

                    if params == 'None':
                        return -999
                        
                    model = get_model(name, params)
                    model.fit(X_train_scaled, y_train)
                    y_pred = model.predict(X_test_scaled)
                    score = r2_score(y_test, y_pred)
                    print(f"Trial {trial.number}: R2 = {score:.4f}")
                    return score
                except Exception as e:
                    return -999
            
            # Create and run Optuna study
            study = optuna.create_study(direction='maximize', sampler=optuna.samplers.TPESampler(seed=42))
            optuna.logging.set_verbosity(optuna.logging.CRITICAL)
            study.optimize(objective, n_trials=n_trials)
            
            # Store best parameters if optimization was successful
            if study.best_value > -999:
                temp = pd.DataFrame([study.best_params])
                best_params_df = pd.concat([best_params_df, temp], ignore_index=True)
                successful_combos += 1
            else:
                failed_combos += 1
                
        except Exception as e:
            print(f"  Error processing combination {i+1}: {str(e)}")
            failed_combos += 1
            continue
    
    print(f"  Processed {successful_combos} combinations successfully, {failed_combos} failed")
    
    # Average parameters across all samples
    if len(best_params_df) == 0:
        print("  WARNING: No successful parameter optimization. Returning empty parameters.")
        return pd.DataFrame({'model': [name], 'hyperparameters': ['{}']})
    
    # Define parameter types for averaging
    int_cols = ['min_data_in_leaf', 'bagging_freq', 'iterations', 'border_count', 
                'n_estimators', 'min_samples_split', 'min_samples_leaf', 'num_leaves', 
                'max_depth', 'depth']
    float_cols = ['learning_rate', 'lambda_l1', 'lambda_l2', 'feature_fraction', 
                  'bagging_fraction', 'subsample', 'l2_leaf_reg', 'bagging_temperature', 
                  'random_strength', 'gamma', 'reg_alpha', 'reg_lambda', 'colsample_bytree']
    text_cols = ['max_features', 'loss', 'grow_policy', 'bootstrap', 'oob_score']
    
    params_avg = {}
    
    for col in best_params_df.columns:
        if np.issubdtype(best_params_df[col].dtype, np.integer):
            params_avg[col] = int(round(best_params_df[col].mean()))
        elif np.issubdtype(best_params_df[col].dtype, np.floating):
            params_avg[col] = best_params_df[col].mean()
        else:  # object / categorical
            params_avg[col] = best_params_df[col].mode()[0]
    
    # Return as DataFrame
    params_avg_df = pd.DataFrame({'model': [name], 'hyperparameters': [str(params_avg)]})
    
    return params_avg_df

def sample_data_for_tuning(df, random_state=42):
    """
    For each model_id, randomly sample one (region_name, feature_set_name) combination.

    Returns:
        List of DataFrames (one for each sampled combination)
    """
    np.random.seed(random_state)

    sampled_data_list = []

    unique_combos = df[['model_name', 'region_name', 'feature_set_name']].drop_duplicates()

    # Sample one combination per model_name
    for model in unique_combos['model_name'].unique():
        model_combos = unique_combos[unique_combos['model_name'] == model]
        if not model_combos.empty:
            sampled_combo = model_combos.sample(1, random_state=random_state).iloc[0]

            # Filter full data for the sampled combo
            combo_df = df[
                (df['model_name'] == sampled_combo['model_name']) &
                (df['region_name'] == sampled_combo['region_name']) &
                (df['feature_set_name'] == sampled_combo['feature_set_name'])
            ].copy()

            if not combo_df.empty:
                print(f"Sampled for model_name: {model} -> region: {sampled_combo['region_name']}, feature_set: {sampled_combo['feature_set_name']}")
                sampled_data_list.append(combo_df)

    return sampled_data_list
def find_best_hyperparameters(df, model_names, n_trials,test_size):
    """
    Find best hyperparameters by sampling data and running Optuna optimization.

    Args:
        df: Prepared dataset (Pandas DataFrame)
        model_names: List of model names to tune (e.g., ['RandomForestRegressor'])
        n_trials: Number of Optuna trials per combination

    Returns:
        DataFrame with best hyperparameters for each model
    """
    # Step 1: Sample one (model_name, feature_set_name) per region
    sampled_data_list = sample_data_for_tuning(df, random_state=42)
    # print(sampled_data_list)

    # Step 2: Extract metadata from sampled combos
    metadata = []
    for combo_df in sampled_data_list:
        if combo_df.empty:
            continue  # Skip empty combos
        row = combo_df.iloc[0]
        metadata.append({
            'region_name': row['region_name'],
            'model_name': row['model_name'],
            'feature_set_name':row['feature_set_name']
        })

    df_sampled_metadata = pd.DataFrame(metadata)

    # Step 3: Combine sampled data into one dataframe
    df_sampled_combined = pd.concat(sampled_data_list, ignore_index=True)

    # Step 4: Initialize result container
    tuned_hyperparameters = pd.DataFrame()

    # Step 5: Tune each model
    for model_name in model_names:
        print(f"\nTuning {model_name}...")

        try:
            result = optuning(
                df_mds=df_sampled_combined,
                df_region_sku_sampled=df_sampled_metadata,

def calculate_elasticities(model, X, method='pdp', price_col='price', discount_col='total_discount_ratio'):
    valid_methods = ['pdp', 'marginal']

    if method not in valid_methods:
        raise ValueError(f"Invalid method. Choose from {valid_methods}")

    if price_col not in X.columns:
        raise ValueError(f"{price_col} not found in the dataset")

    elasticities = {'price_elasticity': None, 'promo_elasticity': None}

    if method == 'pdp':
        if price_col in X.columns:
            pdp_price = partial_dependence(model, X, [X.columns.get_loc(price_col)], grid_resolution=100)
            prices, quantities = pdp_price['values'][0], pdp_price['average'][0]
            
            price_changes = np.gradient(prices)
            quantity_changes = np.gradient(quantities)
            price_elasticities = np.divide(quantity_changes / quantities, price_changes / prices, 
                                        out=np.zeros_like(quantity_changes), where=price_changes != 0)
            elasticities['price_elasticity'] = np.round(np.mean(price_elasticities),3)
        else:
            elasticities['price_elasticity'] = None
    
    return elasticities

def feature_importances(name, model, X_train):
    """
    Returns a dictionary of feature importances for various model types.

    Args:
        name (str): Model name.
        model: Trained model object.
        X_train (pd.DataFrame): Training data features.

    Returns:
        dict: Sorted dictionary of feature importances.
    """
    features = X_train.columns.tolist()

    # if name in ['LinearRegression', 'LogLinear', 'LogLogLinear']:
    #     importances = model.coef_
    if name in ['LinearRegression','LogLinear', 'LogLogLinear', 'LinearLog', 'LinearRegression']:
        importances = model.coef_
    elif name == 'CatBoostRegressor':
        importances = model.get_feature_importance()
    elif name in ['RandomForestRegressor', 'XGBRegressor', 'AdaBoostRegressor']:
        importances = model.feature_importances_
    else:
        importances = np.zeros(len(features))  # Default case

    # Flatten if needed (e.g. if importances is 2D)
    if isinstance(importances, np.ndarray) and importances.ndim > 1:
        importances = importances[0]

    # Map feature names to absolute importances
    importance_dct = {feature: abs(imp) for feature, imp in zip(features, importances)}
    importance_dct = dict(sorted(importance_dct.items(), key=lambda item: item[1], reverse=True))

    return importance_dct

                    'total_volume_actual': y_train.tolist() + y_test.tolist(),
                    'total_volume_pred': y_train_pred.tolist() + y_test_pred.tolist(),
                    'week': weeks_train + weeks_test
                })

                scatter_plot_df = pd.concat([scatter_plot_df, temp], ignore_index=True)

            except Exception as e:
                print(f"Error with {name}: {e}")
                continue

    results[['r2_train', 'r2_test']] = round(results[['r2_train', 'r2_test']], 3)
    return scatter_plot_df, results

final_model_res = spark.sql(f"""
-- create or replace table {final_model_results} as

with cte as 
(
SELECT
    mr.model_name,
    -- mr.material_group_name,
    mr.region_name,
    mr.model as ml_model,
    feature_set_name,
    r2_train,
    round(r2_test,3) as r2,
    price_elasticity,
    Round(rmse,2) as rmse,
    mr.feature_importance,
    -- pm.hyperparameters,
    True as temporal_split_shuffle
FROM {final_iterations_tb} mr
WHERE
    1 = 1
    -- and comment = 'temporal split'
    and r2_train >= 0.7 and r2_train <= 0.97 
    --and ROUND(r2_test,1) >= 0.6
    and r2_test >= 0.6
    and r2_test <= r2_train
    and (r2_train - r2_test) < 0.4
    AND price_elasticity < 0
    and  feature_set_name not ilike '%comp_per_price_%'
    -- and mr.model_name in ('LULU HIMA.PINK SALT JAR','LULU HIMA.PINK SALT POUCH')
    and mr.model_name in ({', '.join([repr(x) for x in model_filter_list])})
)
,cte2 as 
(
select *
from cte
qualify row_number() over (partition by model_name, region_name order by r2 desc) = 1
)
,intermediate as 
(
select cte2.* except(feature_set_name,ml_model),pred.* except(model_name,region_name,price,total_volume_Actual),mds.* except(model_name,region_name,week_number),pm.hyperparameters
from cte2 
join {final_preds_tb} pred
on cte2.model_name = pred.model_name 
    and cte2.region_name = pred.region_name
    and cte2.feature_set_name = pred.feature_set_name
    and cte2.ml_model = pred.model
join {final_mds_table} mds
on mds.model_name = pred.model_name 
    and mds.region_name = pred.region_name
    and mds.week_number = pred.week
left join {best_params_tb} pm
on cte2.ml_model = pm.model
AND mds.material_group_name = pm.material_group 
-- where pred.model_name in ('LULU HIMA.PINK SALT JAR','LULU HIMA.PINK SALT POUCH')
where pred.model_name in ({', '.join([repr(x) for x in model_filter_list])})
)

select * except(ingestion_ts), max(ingestion_ts) as max_ingestion_ts
from intermediate
group by all

""")
