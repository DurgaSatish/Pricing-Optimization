create or replace table dev.sandbox.av_all_ean_seta
with base as 
(
  select sr.* except(category_name,Material_group_name),mm.material_group_name,mm.category_name
  from analytics.pricing.seta_accepted_recommendation_info sr
  left join gold.material.material_master mm 
  on sr.material_id = mm.material_id
)
,ean_info as 
(
select distinct base.*,mem.numerator,mem.denominator
from base  
left join gold.material.material_ean_master mem 
on mem.material_id = base.material_id 
and bigint(base.ean) = bigint(mem.ean)
)
,base_ean as 
(
select * except(insert_date,insert_utc_timestamp,mem.material_id,ean_info.ean,mem.numerator,mem.denominator,mem.material_name)
from ean_info 
left join gold.material.material_ean_master mem 
on mem.material_id = ean_info.material_id 
and ean_info.numerator = mem.numerator 
and ean_info.denominator = mem.denominator
and mem.unit != 'ZU'
)
,implementation as 
(
select base_ean.*except(ean),base_ean.ean as ean_final,coalesce(min(business_day),implementation_date) as implementation_final
from  base_ean 
join gold.store.store_master sm 
on base_ean.region_name =sm.region_name 
left join gold.transaction.uae_pos_transactions pos 
on base_ean.material_id = pos.product_id
-- and base_ean.ean = pos.ean
and pos.store_id = sm.store_id
and business_day >= implementation_date
and regular_unit_price = accepted_price
group by all
)

,cogs as 
(
  select sd.*,sm.region_name,row_number() over (partition by material_id,region_name order by month(business_date) desc) as rn
  from gold.store.store_master sm 
  left join gold.store.stock_details sd 
  on sd.store_id = sm.store_id
  where (material_id,region_name) in (select distinct material_id,region_name from implementation)
  
)
,latest_cogs as 
(
  select material_id,region_name,round(avg(moving_avg_price),2) as base_cost
  from cogs 
  where rn = 1
  group by all
)
,final as 
(
select imp.*,cogs.base_cost as base_cost,round((imp.numerator/imp.denominator)* cogs.base_cost,2) as unit_cost
from implementation imp
left join latest_cogs cogs 
on cogs.material_id = imp.material_id 
and cogs.region_name = imp.region_name
)

select sr.material_id,sr.region_name,sr.Material_name,round(sr.Final_recommnedation,2) as expected_price,ae.* except(ae.material_id,ae.region_name,Material_name,implementation_final,expected_price,ean_final,material_group_name,category_name),coalesce(implementation_final,'2025-08-14') as implementation_final,coalesce(ae.ean_final,sr.ean) as ean_final,mm.material_group_name,mm.category_name
from analytics.pricing.seta_final_recommendation sr
left join final ae

from pyspark.sql import functions as F
from datetime import datetime, timedelta

# If the Implementation Period is 3 months then run the code as it is, or else change the days count value of the variable num_days
# current_date cutoff (1 day prior to today)
current_date = (datetime.today() - timedelta(days=1)).strftime('%Y-%m-%d')

# This below date is for capturing the baseline data, which is the Last 3 months start date used in the final model price recommendation
start_date_pre = '2025-04-20'

# baseline = 12 weeks = 84 days
BASELINE_NUM_DAYS = 104
num_days = 90
df_spark = (
    df
    # Parse dd.MM.yyyy -> date type
    # .withColumn("implementation_final", F.to_date("implementation_final", "dd.MM.yyyy"))
    
    # Compute end_date = implementation_date + 30 days
    .withColumn("end_date", F.date_add(F.col("implementation_final"), num_days))
    
    # Apply current_date cutoff for final_end_date
    .withColumn("final_end_date", F.least(F.col("end_date"), F.to_date(F.lit(current_date), "yyyy-MM-dd")))
    
    # Compute number of implementation days
    .withColumn(
        "num_implementation_days",
        F.datediff(F.col("final_end_date"), F.col("implementation_final")) + F.lit(1)
    )
    
    # ---- PRE PERIOD CALCULATIONS ----
    # Start date of pre-period = 12 weeks before implementation
   .withColumn('start_date_pre', F.lit(f'{start_date_pre}'))
    
    # End date of pre-period (aligned to implementation duration)
    .withColumn("end_date_pre", F.expr("date_add(start_date_pre, num_implementation_days - 1)"))
    
    # End date of overall pre-period baseline (full 84 days)
    .withColumn("end_date_pre_overall", F.date_add(F.col("start_date_pre"), BASELINE_NUM_DAYS))
    
    # Equivalent pre-period length (in days)
    .withColumn("num_equivalent_days", F.datediff(F.col("end_date_pre"), F.col("start_date_pre")) + F.lit(1))
    
    # Standardize dates into yyyy-MM-dd (optional)
    .withColumn("implementation_final", F.date_format("implementation_final", "yyyy-MM-dd"))
    .withColumn("end_date", F.date_format("end_date", "yyyy-MM-dd"))
    .withColumn("final_end_date", F.date_format("final_end_date", "yyyy-MM-dd"))
    .withColumn("start_date_pre", F.date_format("start_date_pre", "yyyy-MM-dd"))
    .withColumn("end_date_pre", F.date_format("end_date_pre", "yyyy-MM-dd"))
    .withColumn("end_date_pre_overall", F.date_format("end_date_pre_overall", "yyyy-MM-dd"))
)

df_spark.display()
df_spark.write.option('overwriteSchema',True).mode('overwrite').saveAsTable(f'{Final_table}')

create or replace table dev.sandbox.av_violation_reasons1
WITH daily_violations AS (
  SELECT 
    material_id,
    store_id,
    material_name,
    region_name,
    business_day,
    observed_price,
    accepted_price
  FROM dev.sandbox.am_test
  WHERE valid_price_flag = 0
),

region_store_counts AS (
  SELECT 
    material_id,
    region_name,
    business_day,
    COUNT(DISTINCT store_id) AS total_stores
  FROM dev.sandbox.am_test
  GROUP BY material_id, region_name, business_day
),

violation_store_counts AS (
  SELECT 
    material_id,
    region_name,
    business_day,
    COUNT(DISTINCT store_id) AS violating_stores
  FROM daily_violations
  GROUP BY material_id, region_name, business_day
),

region_violations AS (
  SELECT 
    v.material_id,
    NULL AS store_id,
    d.material_name,
    v.region_name,
    v.business_day,
    'region' AS violation_type,
    d.observed_price,
    d.accepted_price
  FROM violation_store_counts v
  JOIN region_store_counts r 
    ON v.material_id = r.material_id 
   AND v.region_name = r.region_name 
   AND v.business_day = r.business_day
  JOIN daily_violations d 
    ON v.material_id = d.material_id 
   AND v.region_name = d.region_name 
   AND v.business_day = d.business_day
  WHERE v.violating_stores = r.total_stores
  GROUP BY v.material_id, d.material_name, v.region_name, v.business_day, d.observed_price, d.accepted_price
),

store_violations AS (
  SELECT DISTINCT
    d.material_id,
    d.store_id,
    d.material_name,
    d.region_name,
    d.business_day,
    'store' AS violation_type,
    d.observed_price,
    d.accepted_price
  FROM daily_violations d
  JOIN violation_store_counts v 
    ON d.material_id = v.material_id 
   AND d.region_name = v.region_name 
   AND d.business_day = v.business_day
  JOIN region_store_counts r 
    ON v.material_id = r.material_id 
   AND v.region_name = r.region_name 
   AND v.business_day = r.business_day
  WHERE v.violating_stores < r.total_stores
),

all_violations AS (
  SELECT * FROM region_violations
  UNION ALL
  SELECT * FROM store_violations
),

violation_periods AS (
  SELECT 
    material_id,
    store_id,
    material_name,
    region_name,
    violation_type,
    business_day,
    observed_price,
    accepted_price,
    business_day - ROW_NUMBER() OVER (
      PARTITION BY material_id, COALESCE(store_id, region_name), violation_type 
      ORDER BY business_day
    ) AS grp
  FROM all_violations
),

aggregated_periods AS (
  SELECT 
    material_id,
    store_id,
    material_name,
    region_name,
    violation_type,
    MIN(business_day) AS violation_start_date,
    MAX(business_day) AS violation_end_date,
    AVG(observed_price) AS avg_observed_price,
    AVG(accepted_price) AS avg_expected_price
  FROM violation_periods
  GROUP BY material_id, store_id, material_name, region_name, violation_type, grp
),

reason_mapping AS (
  SELECT 
    material_id,
    region_name,
    store_id,
    Violation_start_date,
    violation_end_date,
    reasons 
  FROM dev.sandbox.av_violation_reasons
)

-- ✅ Final join: map reasons where periods overlap
SELECT 
distinct 
program_name,
    v.*,
    implementation_date,
    r.reasons
FROM aggregated_periods v
LEFT JOIN reason_mapping r
  ON v.material_id = r.material_id
 AND COALESCE(v.store_id, -1) = COALESCE(r.store_id, -1)
 AND v.region_name = r.region_name
 AND (
      -- The violation periods overlap
      v.violation_start_date BETWEEN r.violation_start_date AND r.violation_end_date
      or 
      v.violation_end_date between r.violation_start_date AND r.violation_end_date
      or
       v.violation_end_date BETWEEN r.violation_start_date AND r.violation_end_date
      OR 
      (r.violation_start_date BETWEEN v.violation_start_date AND v.violation_end_date)
    )
left join (select program_name,material_id,region_name,min(implementation_date) as implementation_date 
from dev.sandbox.av_all_ean
group by all ) at 
on at.material_id = v.material_id and at.region_name = v.region_name
left anti join dev.sandbox.av_accepted_file imp 
on imp.material_id = v.material_id and imp.region_name = v.region_name
-- and imp.program_name = 'SETA Pricing August 2025'
ORDER BY v.material_id, v.region_name, v.store_id, v.violation_start_date;

create or replace table dev.sandbox.av_growth_store_region
WITH pre_period_data AS (
    SELECT 
        sm.region_name,
        sm.store_id,
        mm.material_group_name,
        mm.category_name,
        MONTH(p.business_day) AS month,
        SUM(p.quantity) AS total_qty_pre,
        DATEDIFF('2025-07-20', '2025-04-27') AS pre_days
    FROM gold.transaction.uae_pos_transactions p
    JOIN gold.store.store_master sm 
        ON sm.store_id = p.store_id
    JOIN gold.material.material_master mm 
        ON mm.material_id = p.product_id
    WHERE p.business_day BETWEEN '2025-04-27' AND '2025-07-20'
      AND category_name IN (
            SELECT DISTINCT category_name 
            FROM analytics.pricing.seta_final_recommendation
      )
      AND p.amount > 0
      AND p.quantity > 0
      AND p.bulk_sale_flag IS NULL
    GROUP BY sm.region_name, sm.store_id, mm.material_group_name, mm.category_name, MONTH(p.business_day)
),
test_period_data AS (
    SELECT 
        sm.region_name,
        sm.store_id,
        mm.material_group_name,
        mm.category_name,
        MONTH(p.business_day) AS month,
        SUM(p.quantity) AS total_qty_test,
        DATEDIFF(CURRENT_DATE(), '2025-08-12') AS test_days
    FROM gold.transaction.uae_pos_transactions p
    JOIN gold.store.store_master sm 
        ON sm.store_id = p.store_id
    JOIN gold.material.material_master mm 
        ON mm.material_id = p.product_id
    WHERE p.business_day BETWEEN '2025-08-12' AND CURRENT_DATE()-1
      AND p.amount > 0
      AND p.quantity > 0
      AND p.bulk_sale_flag IS NULL
    GROUP BY sm.region_name, sm.store_id, mm.material_group_name, mm.category_name, MONTH(p.business_day)
),
deseasonalized_pre_period AS (
    SELECT 
        pre.region_name,
        pre.store_id,
        pre.material_group_name,
        pre.category_name,
        pre.month,
        pre.total_qty_pre,
        pre.pre_days,
        ROUND(pre.total_qty_pre / NULLIF(m.material_group_seasonality, 0), 2) AS deseasonalized_qty_pre
    FROM pre_period_data pre
    JOIN dev.sandbox.av_mg_cat_seasonality1 m
      ON pre.region_name = m.region_name
     AND pre.material_group_name = m.material_group_name
     AND pre.category_name = m.category_name
     AND pre.month = m.month
),
deseasonalized_test_period AS (
    SELECT 
        test.region_name,
        test.store_id,
        test.material_group_name,
        test.category_name,
        test.month,
        test.total_qty_test,
        test.test_days,
        ROUND(test.total_qty_test / NULLIF(m.material_group_seasonality, 0), 2) AS deseasonalized_qty_test
    FROM test_period_data test
    JOIN dev.sandbox.av_mg_cat_seasonality1 m
      ON test.region_name = m.region_name
     AND test.material_group_name = m.material_group_name
     AND test.category_name = m.category_name
     AND test.month = m.month
),
pre_day_deseasonalized AS (
  SELECT 
      region_name,
      store_id,
      material_group_name,
      category_name,
      SUM(deseasonalized_qty_pre) / MAX(pre_days) AS pre_per_day
  FROM deseasonalized_pre_period
  GROUP BY region_name, store_id, material_group_name, category_name
),
test_day_deseasonalized AS (
  SELECT 
      region_name,
      store_id,
      material_group_name,
      category_name,
      SUM(deseasonalized_qty_test) / MAX(test_days) AS test_per_day
  FROM deseasonalized_test_period
  GROUP BY region_name, store_id, material_group_name, category_name
),
-- Store + Material Group level growth
mg_growth AS (
  SELECT 
      a.store_id,
      a.region_name,
      a.category_name,
      a.material_group_name,
      ROUND((b.test_per_day - a.pre_per_day) / NULLIF(a.pre_per_day,0),4) AS mg_growth_store
  FROM pre_day_deseasonalized a
  LEFT JOIN test_day_deseasonalized b
    ON a.region_name = b.region_name
   AND a.store_id = b.store_id 
   AND a.material_group_name = b.material_group_name
   AND a.category_name = b.category_name
),
-- Store + Category level growth (aggregate across material groups)
cat_growth AS (
  SELECT 
      a.store_id,
      a.region_name,
      a.category_name,
      ROUND(
        (SUM(b.test_per_day) - SUM(a.pre_per_day)) / NULLIF(SUM(a.pre_per_day),0),
        4
      ) AS cat_growth_store
  FROM pre_day_deseasonalized a
  LEFT JOIN test_day_deseasonalized b
    ON a.region_name = b.region_name
   AND a.store_id = b.store_id 
   AND a.material_group_name = b.material_group_name
   AND a.category_name = b.category_name
  GROUP BY a.store_id, a.region_name, a.category_name
),
-- Region + Material Group level growth
region_mg_growth AS (
  SELECT 
      a.region_name,
      a.category_name,
      a.material_group_name,
      ROUND(
        (SUM(b.test_per_day) - SUM(a.pre_per_day)) / NULLIF(SUM(a.pre_per_day),0),
        4
      ) AS mg_growth_region
  FROM pre_day_deseasonalized a
  LEFT JOIN test_day_deseasonalized b
    ON a.region_name = b.region_name
   AND a.store_id = b.store_id 
   AND a.material_group_name = b.material_group_name
   AND a.category_name = b.category_name
  GROUP BY a.region_name, a.category_name, a.material_group_name
),
-- Region + Category level growth (aggregate across material groups and stores)
region_cat_growth AS (
  SELECT 
      a.region_name,
      a.category_name,
      ROUND(
        (SUM(b.test_per_day) - SUM(a.pre_per_day)) / NULLIF(SUM(a.pre_per_day),0),
        4
      ) AS cat_growth_region
  FROM pre_day_deseasonalized a
  LEFT JOIN test_day_deseasonalized b
    ON a.region_name = b.region_name
   AND a.store_id = b.store_id 
   AND a.material_group_name = b.material_group_name
   AND a.category_name = b.category_name
  GROUP BY a.region_name, a.category_name
)

-- Final Output: store growth + region growth
SELECT 
    m.store_id,
    m.region_name,
    m.category_name,
    m.material_group_name,
    m.mg_growth_store,
    c.cat_growth_store,
    rm.mg_growth_region,
    rc.cat_growth_region
FROM mg_growth m
JOIN cat_growth c
  ON m.store_id = c.store_id
 AND m.region_name = c.region_name
 AND m.category_name = c.category_name
JOIN region_mg_growth rm
  ON m.region_name = rm.region_name
 AND m.category_name = rm.category_name
 AND m.material_group_name = rm.material_group_name
JOIN region_cat_growth rc
  ON m.region_name = rc.region_name
 AND m.category_name = rc.category_name

WITH valid_products AS (
    SELECT DISTINCT 
        material_id product_id, 
        region_name,
        ean_final
    FROM dev.sandbox.av_all_ean
    WHERE accepted_price IS NOT NULL
),

store_price AS (
    SELECT
        business_day,
        sm.region_name,
        pos.store_id,
        pos.product_id,
        SUM(amount) AS total_sales,
        SUM(quantity) AS total_qty,
        SUM(amount) / SUM(quantity) AS store_price
    FROM gold.transaction.uae_pos_transactions pos
    JOIN gold.store.store_master sm 
        ON sm.store_id = pos.store_id
    JOIN valid_products vp 
        ON vp.product_id = pos.product_id
       AND vp.region_name = sm.region_name
    WHERE business_day BETWEEN '2025-08-28' AND '2025-11-26'
    and ean in (select distinct ean_final from dev.sandbox.av_all_ean)
      AND amount > 0
      AND quantity > 0
    GROUP BY business_day, sm.region_name, pos.store_id, pos.product_id
),

region_price AS (
    SELECT
        business_day,
        region_name,
        product_id,
        SUM(total_sales) / SUM(total_qty) AS region_avg_price
    FROM store_price
    GROUP BY business_day, region_name, product_id
),

comparison AS (
    SELECT
        sp.business_day,
        sp.region_name,
        sp.product_id,
        sp.store_id,
        CASE
            WHEN sp.store_price BETWEEN rp.region_avg_price * 0.95
                                   AND rp.region_avg_price * 1.05
            THEN 0
            ELSE 1
        END AS violation_flag
    FROM store_price sp
    JOIN region_price rp
      ON sp.business_day = rp.business_day
     AND sp.region_name = rp.region_name
     AND sp.product_id = rp.product_id
),

store_violation AS (
    SELECT
        region_name,
        product_id,
        store_id,
        MAX(violation_flag) AS store_violation
    FROM comparison
    GROUP BY region_name, product_id, store_id
)

SELECT
    region_name,
    product_id,
    SUM(store_violation) AS stores_outside_5pct,
    COUNT(*) AS total_stores,
    ROUND(SUM(store_violation) / COUNT(*), 4) AS violation_ratio
FROM store_violation
GROUP BY region_name, product_id
ORDER BY product_id, region_name;
