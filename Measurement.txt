from pyspark.sql import functions as F
from datetime import datetime, timedelta

# current_date cutoff (1 day prior to today)
current_date = (datetime.today() - timedelta(days=1)).strftime('%Y-%m-%d')
start_date_pre = '2025-04-20'

# baseline = 12 weeks = 84 days
BASELINE_NUM_DAYS = 104

df_spark = (
    df
    # Parse dd.MM.yyyy -> date type
    # .withColumn("implementation_final", F.to_date("implementation_final", "dd.MM.yyyy"))
    
    # Compute end_date = implementation_date + 90 days
    .withColumn("end_date", F.date_add(F.col("implementation_final"), 90))
    
    # Apply current_date cutoff for final_end_date
    .withColumn("final_end_date", F.least(F.col("end_date"), F.to_date(F.lit(current_date), "yyyy-MM-dd")))
    
    # Compute number of implementation days
    .withColumn(
        "num_implementation_days",
        F.datediff(F.col("final_end_date"), F.col("implementation_final")) + F.lit(1)
    )
    
    # ---- PRE PERIOD CALCULATIONS ----
    # Start date of pre-period = 12 weeks before implementation
   .withColumn('start_date_pre', F.lit(f'{start_date_pre}'))
    
    # End date of pre-period (aligned to implementation duration)
    .withColumn("end_date_pre", F.expr("date_add(start_date_pre, num_implementation_days - 1)"))
    
    # End date of overall pre-period baseline (full 84 days)
    .withColumn("end_date_pre_overall", F.date_add(F.col("start_date_pre"), BASELINE_NUM_DAYS))
    
    # Equivalent pre-period length (in days)
    .withColumn("num_equivalent_days", F.datediff(F.col("end_date_pre"), F.col("start_date_pre")) + F.lit(1))
    
    # Standardize dates into yyyy-MM-dd (optional)
    .withColumn("implementation_final", F.date_format("implementation_final", "yyyy-MM-dd"))
    .withColumn("end_date", F.date_format("end_date", "yyyy-MM-dd"))
    .withColumn("final_end_date", F.date_format("final_end_date", "yyyy-MM-dd"))
    .withColumn("start_date_pre", F.date_format("start_date_pre", "yyyy-MM-dd"))
    .withColumn("end_date_pre", F.date_format("end_date_pre", "yyyy-MM-dd"))
    .withColumn("end_date_pre_overall", F.date_format("end_date_pre_overall", "yyyy-MM-dd"))
)

df_spark.display()

-- Replace the holiday_list table with the latest table
create or replace table dev.sandbox.av_promo_tag as 
with region_wise as
(
select period_type,material_id,business_day,region_name,material_group_name,sum(num_txn) as num_txn,round(sum(total_sales),2) as total_sales,sum(total_qty) as total_qty,round(avg(rd.avg_regular_unit_price),2) as avg_regular_unit_price,max(case when hl.holiday_date is not null then 1 else 0 end) as holiday
from dev.sandbox.pricing_measurement_raw_txn_data rd
left join dev.sandbox.pg_uae_holiday_list hl
on rd.business_day = hl.holiday_date


group by all
)
,base AS (
SELECT
*,
case when day(business_day) in (27,28,29,30,31,1,2,3,4,5) then 'PAY DAY' else 'NON PAY DAY' end as week_type,
ROUND(total_sales / NULLIF(total_qty,0), 2) AS unit_price_derived,
LAG(ROUND(total_sales / NULLIF(total_qty,0), 2)) OVER (
PARTITION BY region_name, material_id
ORDER BY business_day
) AS past_price,
(ROUND(total_sales / NULLIF(total_qty,0), 2) /
LAG(ROUND(total_sales / NULLIF(total_qty,0), 2)) OVER (
PARTITION BY region_name, material_id
ORDER BY business_day
) - 1) AS change_percentage
FROM region_wise
-- WHERE material_id = 1815319
-- AND region_name = 'DUBAI'
),
direction_flag AS (
SELECT *,
CASE
WHEN period_type = 'pre_period' and change_percentage > 0.05 THEN 'Increase'
WHEN period_type = 'pre_period' and change_percentage < -0.08 THEN 'Decrease'
END AS price_direction
FROM base
),

events AS (
SELECT
region_name, material_id, business_day, price_direction,period_type,
ROW_NUMBER() OVER (PARTITION BY region_name, material_id ORDER BY business_day) AS rn,
MIN(CASE WHEN price_direction = 'Increase' THEN business_day END) OVER (PARTITION BY region_name, material_id ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING) AS max_inc_day1
FROM direction_flag
WHERE price_direction IS NOT NULL
),


max_inc_day AS (

  select 
    *,
    case when max_inc_day1 is NULL then max(business_day) over() else max_inc_day1 end as max_inc_day_new
  from events
),

valid_pairs AS (
SELECT *,
DATEDIFF(max_inc_day_new, business_day) AS gap
FROM max_inc_day
WHERE DATEDIFF(max_inc_day_new, business_day) <= 21
AND price_direction = 'Decrease'
)

,final as
(
SELECT
b.*,
CASE
WHEN vp.material_id IS NOT NULL or pi.material_id is not null THEN 1 ELSE 0
END AS promo_flag
FROM base b
LEFT JOIN valid_pairs vp
ON b.region_name = vp.region_name
AND b.material_id = vp.material_id
-- AND b.business_day BETWEEN vp.dec_day AND vp.inc_day
AND b.business_day >= vp.business_day 
AND b.business_day <= vp.max_inc_day_new
left join dev.sandbox.av_promo_identified pi
on pi.material_id = b.material_id
and pi.region_name = b.region_name
and b.business_day between pi.actual_start_date and pi.actual_end_date
)

select 
distinct * 
from final
-- where material_id = 1685342 AND region_name = 'DUBAI'
-- where promo_flag = 0
-- and holiday = 0
where promo_flag = 0
and holiday = 0

CREATE OR REPLACE TEMP VIEW test_skus_comparative_view AS 
(
  SELECT DISTINCT
    COALESCE(pt.program_name, tp.program_name) AS program_name,
    COALESCE(pt.material_group_name, tp.material_group_name) AS material_group_name,
    COALESCE(pt.category_name, tp.category_name) AS category_name,
    COALESCE(pt.region_name, tp.region_name) AS region_name,
    COALESCE(pt.store_id, tp.store_id) AS store_id,
    COALESCE(pt.material_id, tp.material_id) AS material_id,
    COALESCE(pt.total_qty_raw_pre, 0) AS total_qty_raw_pre,
    COALESCE(pt.num_txn_pre, 0) AS num_txn_pre,
    COALESCE(pt.num_days_pre, 0) AS num_days_pre,
    COALESCE(pt.total_qty_pre, 0) AS total_qty_pre,
    COALESCE(pt.days_preperiod, 0) AS days_preperiod,
    COALESCE(pt.total_sales_pre, 0) AS total_sales_pre,
    COALESCE(pt.avg_regular_unit_price_pre, 0) AS avg_regular_unit_price_pre,
    COALESCE(pt.avg_unit_price_derived_pre, 0) AS avg_unit_price_derived_pre,
    COALESCE(pt.p20_qty, 0) AS p20_qty,
    COALESCE(pt.median_qty, 0) AS median_qty,
    COALESCE(pt.p90_qty, 0) AS p90_qty,
    COALESCE(pt.daily_qty_raw, 0) AS daily_qty_raw,
    COALESCE(pt.avg_qty, 0) AS avg_qty_pre,
    COALESCE(pt.daily_qty, 0) AS pre_deseasonalized_qty,

    COALESCE(tp.num_txn, 0) AS num_txn,
    COALESCE(tp.num_days, 0) AS num_days,
    COALESCE(tp.total_qty_raw, 0) AS test_raw_qty,
    COALESCE(tp.avg_qty, 0) AS avg_qty,
    COALESCE(tp.total_qty, 0) AS test_deseasonalized_qty,
    COALESCE(tp.total_sales, 0) AS total_sales_test_raw,
    COALESCE(tp.avg_regular_unit_price, 0) AS avg_regular_unit_price,
    COALESCE(tp.avg_unit_price_derived, 0) AS avg_test_unit_price_derived,

    t3.price_change_direction,
    COALESCE(pt.day_type,tp.day_type) as day_type,
    COALESCE(
      CASE WHEN pt.day_type = 'PAY DAY' THEN ri.pay_days ELSE ri.np_days END,
      implementation_days
    ) AS implementation_days,
    ss.valid_counts,
    ss.valid_price_perc

  FROM dev.sandbox.av_pre_txn_info pt

  FULL OUTER JOIN dev.sandbox.av_test_period tp
    ON pt.material_id = tp.material_id
   AND pt.store_id = tp.store_id
   AND pt.day_type = tp.day_type
   AND pt.region_name = tp.region_name

  FULL OUTER JOIN (
    SELECT 
      material_id,
      region_name,
      MIN(pay_days) AS pay_days,
      MIN(np_days) AS np_days
      --, MIN(implementation_days) AS implementation_days
    FROM analytics.pricing.seta_accepted_recommendation_info_new 
    GROUP BY material_id, region_name
  ) ri
    ON COALESCE(pt.material_id, tp.material_id) = ri.material_id
   AND COALESCE(pt.region_name, tp.region_name) = ri.region_name

   JOIN (select distinct material_id,region_name,store_id,price_change_direction from dev.sandbox.all_stores_view) t3
    ON COALESCE(pt.material_id, tp.material_id) = t3.material_id
   AND COALESCE(pt.region_name, tp.region_name) = t3.region_name
   AND COALESCE(pt.store_id, tp.store_id) = t3.store_id

  FULL OUTER JOIN (
    SELECT 
      material_id,
      store_id,
      AVG(valid_counts) AS valid_counts,
      AVG(valid_price_perc) AS valid_price_perc
    FROM dev.sandbox.av_store_sku_violations 
    GROUP BY material_id, store_id
  ) ss
    ON COALESCE(pt.material_id, tp.material_id) = ss.material_id
   AND COALESCE(pt.store_id, tp.store_id) = ss.store_id
)

    (agg_final['avg_test_unit_price_derived'] > agg_final['avg_unit_price_derived_pre']) &

     AND e2.business_day > e1.business_day
    GROUP BY e1.region_name, e1.material_id, e1.business_day,e1.period_type

valid_pairs AS (

%sql
with region_wise as
(
select period_type,material_id,business_day,region_name,material_group_name,sum(num_txn) as num_txn,round(sum(total_sales),2) as total_sales,sum(total_qty) as total_qty,round(avg(rd.avg_regular_unit_price),2) as avg_regular_unit_price,max(case when hl.holiday_date is not null then 1 else 0 end) as holiday
from dev.sandbox.pricing_measurement_raw_txn_data rd
left join dev.sandbox.pg_uae_holiday_list hl
on rd.business_day = hl.holiday_date


group by all
)
,base AS (
SELECT
*,
case when day(business_day) in (27,28,29,30,31,1,2,3,4,5) then 'PAY DAY' else 'NON PAY DAY' end as week_type,
ROUND(total_sales / NULLIF(total_qty,0), 2) AS unit_price_derived,
LAG(ROUND(total_sales / NULLIF(total_qty,0), 2)) OVER (
PARTITION BY region_name, material_id
ORDER BY business_day
) AS past_price,
(ROUND(total_sales / NULLIF(total_qty,0), 2) /
LAG(ROUND(total_sales / NULLIF(total_qty,0), 2)) OVER (
PARTITION BY region_name, material_id
ORDER BY business_day
) - 1) AS change_percentage
FROM region_wise
-- WHERE material_id = 1815319
-- AND region_name = 'DUBAI'
),
direction_flag AS (
SELECT *,
CASE
WHEN period_type = 'pre_period' and change_percentage > 0.05 THEN 'Increase'
WHEN period_type = 'pre_period' and change_percentage < -0.08 THEN 'Decrease'
END AS price_direction
FROM base
),

events AS (
SELECT
region_name, material_id, business_day, price_direction,period_type,
ROW_NUMBER() OVER (PARTITION BY region_name, material_id ORDER BY business_day) AS rn,
MIN(CASE WHEN price_direction = 'Increase' THEN business_day END) OVER (PARTITION BY region_name, material_id ROWS BETWEEN CURRENT ROW AND UNBOUNDED FOLLOWING) AS max_inc_day1
FROM direction_flag
WHERE price_direction IS NOT NULL
),


max_inc_day AS (

  select 
    *,
    case when max_inc_day1 is NULL then max(business_day) over() else max_inc_day1 end as max_inc_day_new
  from events
),

valid_pairs AS (
SELECT *,
DATEDIFF(max_inc_day_new, business_day) AS gap
FROM max_inc_day
WHERE DATEDIFF(max_inc_day_new, business_day) <= 21
AND price_direction = 'Decrease'
)
