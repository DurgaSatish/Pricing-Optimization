# Function to generate dates for a given year, excluding Feb 28 and Feb 29
def generate_dates(year):
    dates = pd.date_range(start=f"{year}-01-01", end=f"{year}-12-31", freq='D')
    return dates[~((dates.month == 2) & (dates.day.isin([28, 29])))]

# Generate date series for all years
dates_2022 = pd.Series(generate_dates(2022))
dates_2023 = pd.Series(generate_dates(2023))
dates_2024 = pd.Series(generate_dates(2024))
dates_2025 = pd.Series(generate_dates(2025))

# Determine the maximum length
max_len = builtins.max(len(dates_2022), len(dates_2023), len(dates_2024), len(dates_2025))

# Reindex to make all series the same length
dates_2022 = dates_2022.reindex(range(max_len))
dates_2023 = dates_2023.reindex(range(max_len))
dates_2024 = dates_2024.reindex(range(max_len))
dates_2025 = dates_2025.reindex(range(max_len))

# Create the wide-format calendar
calendar_df = pd.DataFrame({
    "2022": dates_2022,
    "2023": dates_2023,
    "2024": dates_2024,
    "2025": dates_2025
})

calendar_df["week_number"] = (calendar_df.index // 7) + 1

# Melt to long format
melted_df = pd.melt(calendar_df, id_vars="week_number",
                    value_vars=["2022", "2023", "2024", "2025"],
                    var_name="year", value_name="date")

# Drop rows with missing dates
melted_df = melted_df.dropna(subset=["date"])

# Add week label and format date
melted_df["week_label"] = melted_df.apply(
    lambda row: f"{row['year']}-W{int(row['week_number']):02}", axis=1)

melted_df["date"] = melted_df["date"].dt.strftime('%Y-%m-%d')

# Compute week_start_date per (year, week_number)
week_start = (
    melted_df.groupby(["year", "week_number"])["date"]
    .min()
    .reset_index()
    .rename(columns={"date": "week_start_date"})
)

# Merge back
final_df = (
    melted_df.merge(week_start, on=["year", "week_number"], how="left")
    [["date", "week_label", "week_start_date"]]
    .sort_values(["week_label", "date"])
    .reset_index(drop=True)
)

final_df
